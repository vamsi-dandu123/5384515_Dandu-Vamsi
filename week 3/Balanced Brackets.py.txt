def isBalanced(s):
    # Write your code here
    stack = []
    brackets = {')': '(', '}': '{', ']': '['}
    
    for char in s:
        if char in "({[":
            stack.append(char)
        else:
            if not stack or stack[-1] != brackets[char]:
                return "NO"
            stack.pop()
    return "YES" if not stack else "NO"
