def isValid(s):
    # Write your code here
    freq = Counter(s)
    freq_counts = Counter(freq.values())
    
    if len(freq_counts) == 1:
        return "YES"
    elif len(freq_counts) == 2:
        (f1,c1),(f2,c2) = freq_counts.items()
        if (f1 == 1 and c1 == 1) or (f2 == 1 and c2 == 1):
            return "YES"
        if abs(f1 - f2) == 1 and (c1 == 1 or c2 == 1):
            return "YES"
        return "NO"
    else:
        return "NO"
