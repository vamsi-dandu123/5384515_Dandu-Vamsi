def reverse(llist):
    # Write your code here
    if llist is None or llist.next is None:
        return llist
        
    current = llist
    prev = None
    
    
    while current:
        current.prev,current.next = current.next,current.prev
        
        prev = current
        
        current = current.prev
        
    return prev
