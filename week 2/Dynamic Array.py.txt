def dynamicArray(n, queries):
    # Write your code here
    arr =[[]for _ in range(n)]
    lastAnswer = 0
    answers = []
    
    
    for query in queries:
        t,x,y = query
        idx = (x ^ lastAnswer) % n
        
        if t ==1:
            arr[idx].append(y)
        elif t ==2:
            lastAnswer=arr[idx][y % len(arr[idx])]
            answers.append(lastAnswer)
    return answers
