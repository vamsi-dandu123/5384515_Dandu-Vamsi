def counterGame(n):
    moves = 0
    # Write your code here
    while n > 1:
        if (n & (n-1)) == 0:
            n //= 2
        else:
            highest_power = 1 << (n.bit_length()-1)
            n -= highest_power
        moves += 1
    return "Louise" if moves % 2 == 1 else "Richard"
