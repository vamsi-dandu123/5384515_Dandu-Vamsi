def superDigit(n, k):
    # Write your code here
    digit_sum = sum(int(d) for d in n)
    total = digit_sum * k
    
    
    def helper(x):
        if x < 10:
            return x
        return helper(sum(int(d) for d in str(x)))
    return helper(total)
