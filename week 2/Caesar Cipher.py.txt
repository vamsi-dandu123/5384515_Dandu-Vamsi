def caesarCipher(s, k):
    # Write your code here
    result = []
    k = k % 26
    
    for char in s:
        if char.isalpha():
            if char.islower():
                shifted = chr((ord(char) - ord('a') + k) % 26 + ord('a'))
                result.append(shifted)
            else:
                
                shifted = chr((ord(char)-ord('A')+k)% 26 + ord('A'))
                result.append(shifted)
        else:
            result.append(char)
    return ''.join(result)
