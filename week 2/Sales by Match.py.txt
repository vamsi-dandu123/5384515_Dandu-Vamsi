def sockMerchant(n, ar):
    # Write your code here
    color_count ={}
    pairs = 0
    
    
    for sock in ar:
        color_count[sock] = color_count.get(sock,0)+1
    for count in color_count.values():
        pairs += count // 2
    return pairs
