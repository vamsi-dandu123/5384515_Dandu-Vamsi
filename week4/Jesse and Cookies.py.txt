import heapq
def cookies(k, A):
    # Write your code here
    heapq.heapify(A) 
    operations = 0
    
    while A and A[0] < k:
        if len(A) <2:
            return -1 
            
            
        least = heapq.heappop(A) 
        second_least = heapq.heappop(A) 
        
        new_cookie = least + 2 * second_least
        heapq.heappush(A,new_cookie)
        
        operations += 1
    return operations
