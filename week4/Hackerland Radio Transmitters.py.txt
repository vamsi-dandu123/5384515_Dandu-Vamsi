def hackerlandRadioTransmitters(x, k):

    # Write your code here

    x.sort()

    n= len(x) 

    I =0

    transmitters = 0

    

    

    while I<n:

        transmitters +=1

        

        loc =x[I]+k 

        while I<n and x[I] <= loc:

            I += 1

            

        transmitters_pos = x[I-1]

        

        

        loc = transmitters_pos +k

        while I<n and x[I] <= loc:

            I+=1

    return transmitters
