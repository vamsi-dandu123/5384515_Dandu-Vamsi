cntr = 0

class Node:
    def _init_(self, freq, data):
        self.freq = freq
        self.data = data
        self.left = None
        self.right = None
        global cntr
        self._count = cntr
        cntr = cntr + 1
        
    def _lt_(self, other):
        if self.freq != other.freq:
            return self.freq < other.freq
        return self._count < other._count
