def timeConversion(s):
     period = s[-2:]
     hour = int(s[:2])
     minutes_seconds = s[2:-2]
     
     if period == "AM":
        if hour == 12:
            hour = 0
     else:
        if hour != 12:
            hour += 12
     return "{:02d}{}".format(hour,minutes_seconds)
