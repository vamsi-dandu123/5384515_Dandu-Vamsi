def pangrams(s):
    s = s.lower()
    alphabets = set('abcdefghijklmnopqrstuvwxyz')
    letters_in_s = set(s)
    
    if alphabets.issubset(letters_in_s):
        return "pangram"
    else:
        return "not pangram"
